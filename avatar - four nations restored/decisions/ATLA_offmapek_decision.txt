	rc_increase_influence = {
		only_rulers = yes
		button_name = offmap_ek_inf
		ai_check_interval = 120

		from_potential = {
			has_dlc = "Jade Dragon"
			always = yes
		}
		
		potential = {
			#ROOT = {
			#is_offmap_tag = offmap_ek
			#}
			always = yes
		}

		allow = {
			show_only_failed_conditions = yes
			FROM = {
				show_scope_change = no
				OR = {
					has_landed_title = e_earth_kingdom 
					AND = {
						has_landed_title = k_ba_sing_se
						NOT = { is_title_active = e_earth_kingdom }
					}
				}
				NOT = { is_offmap_governor = offmap_ek }
				check_variable = { which = global_royalAuthority value >= 0 }
				NOT = { has_character_modifier = chinese_grace_send_gift_timer }
			}
		}
		
		effect = {
			FROM = {
				change_variable = { which = global_royalAuthority value = -10 }
				add_offmap_currency = {
					offmap = offmap_ek
					value = 150
				}
			}
			hidden_effect = {
				add_character_modifier = {
					hidden = yes
					name = chinese_grace_send_gift_timer
					days = 180 #Change chinese_grace_send_gift_allow_tt if this number changes
				} 
			}
		}
		
		revoke_allowed = {
			always = no
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0.8
				check_variable = { which = global_royalAuthority value >= 80 }
			}
			modifier = {
				factor = 0.6
				check_variable = { which = global_royalAuthority value >= 60 }
				NOT = { check_variable = { which = global_royalAuthority value >= 80 } }
			}
			modifier = {
				factor = 0.4
				check_variable = { which = global_royalAuthority value >= 40 }
				NOT = { check_variable = { which = global_royalAuthority value >= 60 } }
			}			
			modifier = {
				factor = 0.2
				check_variable = { which = global_royalAuthority value >= 20 }
				NOT = { check_variable = { which = global_royalAuthority value >= 40 } }
			}
			modifier = {
				factor = 0
				check_variable = { which = global_royalAuthority value >= 10 }
				NOT = { check_variable = { which = global_royalAuthority value >= 20 } }
			}			
		}
	}
