# Society Events
namespace = MNHHN

# Hidden start: Send the rank-up request to the current Grandmaster (delayed event on self, from decision to rank up)
character_event = {
	id = MNHHN.3000

	hide_window = yes

	trigger = {
		has_character_flag = has_sent_request_to_rank_up
	}
	
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { is_in_society = yes }
			random_society_member = {
				limit = {
					is_society_grandmaster = yes
					same_society_as = ROOT
				}
				character_event = { id = MNHHN.3001 } # send request to Grand Master. Relic rewards also handled by this event
			}
		}
		if = {
			limit = { is_in_society = no }
			clr_character_flag = has_sent_request_to_rank_up
		}
	}
}

# Grandmaster gets the rank-up request
character_event = {
	id = MNHHN.3001
	
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		event_target:rank_up_target = { letter_event = { id = MNHHN.3002 } }
	}
}

# Force join AI characters to societies
character_event = {
	id = MNHHN.3050
	hide_window = yes
	
	only_rulers = yes
	
	immediate = {
		if = {
			limit = {
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				is_in_society = yes
				society_rank < 3
				society_can_rank_up = yes
			}
			random_list = {
				75 = {}
				25 = {
					society_rank_up = 1					
				}
			}
		}
		if = {
			limit = {
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				is_adult = yes
				is_in_society = no
				controls_religion = no
				block_general_event_trigger = no
			}
			character_event = { id = MNHHN.3051 }
		}
		any_courtier = {
			limit = {
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				is_in_society = yes
				society_rank < 3
				society_can_rank_up = yes
			}
			random_list = {
				95 = {}
				5 = {
					society_rank_up = 1					
				}
			}
		}
		any_courtier = {
			limit = {
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				is_in_society = no
				controls_religion = no
				block_general_event_trigger = no
			}
			character_event = { id = MNHHN.3051 }
		}
	}
}

character_event = {
	id = MNHHN.3051
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		random_list = {
			95 = {}
			5 = {
				trigger = {
					can_join_society = healing_hut_north
				}
				modifier = {
					factor = 3
					OR = {
						trait = waterbender
						trait = kind
						trait = diligent
						trait = temperate
						trait = charitable
						trait = patient
						learning = 12
					}
				}
				modifier = {
					factor = 2
					healing_hut_north = { num_of_society_members < 15 }
				}
				modifier = {
					factor = 2
					healing_hut_north = { num_of_society_members < 10 }
				}
				join_society = healing_hut_north
				if = {
					limit = {
						NOT = {
							society = {
								any_society_member = {
									is_society_grandmaster = yes
								}
							}
						}
					}
					set_society_grandmaster = yes
				}
			}
		}
	}
}
